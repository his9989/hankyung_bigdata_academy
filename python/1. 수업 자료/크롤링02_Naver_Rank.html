def getRealRank(idx):
    url  = 'https://search.naver.com/search.naver?where=nexearch&sm=top_hty&fbm=1&ie=utf8&query=%EC%8B%A4%EC%8B%9C%EA%B0%84+%EA%B2%80%EC%83%89%EC%96%B4+%EC%88%9C%EC%9C%84'
    html = urlopen(url)
    soup = BeautifulSoup(html, "lxml")

    tmp = soup.find_all('ol', 'lst_realtime_srch _tab_area')
    
    if idx >= len(tmp):
        return 'Index Error', ' '
    
    tmp2 = tmp[idx].find_all('li')

    ranks = []
    words = []

    for text in tmp2:
        rank = text.find('em').get_text()
        word = text.find('span', 'tit').get_text()

        ranks.append(rank)
        words.append(word)
        
    return ranks, words
